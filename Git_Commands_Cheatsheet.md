# Git Commands Cheatsheet

## Основные команды
### Инициализация и конфигурация
```bash
git init
# Инициализирует новый локальный репозиторий в текущей папке.

git config --global user.name "Твое_Имя"
# Устанавливает имя пользователя для всех репозиториев.

git config --global user.email "твой_email@example.com"
# Устанавливает email пользователя для всех репозиториев.

Работа с файлами
git add имя_файла
# Добавляет файл в область подготовки (staging area).

git commit -m "Сообщение"
# Фиксирует изменения в репозитории с комментарием.

git status
# Показывает состояние репозитория (отслеживаемые/неотслеживаемые файлы).

git log
# Отображает историю коммитов.

Удаленные репозитории
git remote add origin URL_репозитория
# Добавляет удаленный репозиторий с именем "origin".

git push -u origin main
# Отправляет изменения в удаленный репозиторий в ветку main.

Получение изменений
git pull origin main
# Загружает и сливает изменения из удаленного репозитория.

Полезные команды
pwd
# Печатает текущую директорию.

ls
# Показывает файлы и папки в текущей директории.

cd путь_к_папке
# Переходит в указанную папку.

rm имя_файла
# Удаляет указанный файл.

mv старое_имя новое_имя
# Переименовывает файл или перемещает его.

Настройка SSH
ssh-keygen -t ed25519 -C "твой_email@example.com"
# Генерирует SSH-ключ.

eval "$(ssh-agent -s)"
# Запускает SSH-агент.

ssh-add ~/.ssh/id_ed25519
# Добавляет SSH-ключ в агент.

cat ~/.ssh/id_ed25519.pub
# Показывает публичный SSH-ключ для копирования.
